This demo uses ACE intellisense. Declaration trigger on the keys `<` or `&`. Different documentation is shown for each declaration. The data shown is hard coded.

// setup the ACE editor with html mode var AceIntellisense = ace.require('ace/intellisense').AceIntellisense; var editor5 = ace.edit('editor5'); editor5.getSession().setMode("ace/mode/html"); editor5.focus(); var intellisense5 = new AceIntellisense(editor5); intellisense5.getDecls().setFilterMode('contains'); intellisense5.addDeclarationTrigger({ keyCode: 55, shiftKey: true }); intellisense5.addDeclarationTrigger({ keyCode: 188, shiftKey: true }); intellisense5.addDeclarationTrigger({ keyCode: 190, shiftKey: true }); intellisense5.onDeclaration(function (item) { if (item.keyCode === 190) { intellisense5.getDecls().setVisible(false); } else if (item.keyCode === 55) { var codes = ['nbsp', 'lt', 'gt', 'cent', 'pound', 'curren', 'yen', 'brvbar', 'sect', 'uml', 'copy', 'ordf']; var data = codes.map(function (code) { return { glyph: 5, name: '&' + code + '; (&' + code + ')', value: code + ';' }; }); intellisense5.setDeclarations(data); } else { var data = [ { glyph: 4, name: 'body' }, { glyph: 4, name: 'div' }, { glyph: 4, name: 'head' }, { glyph: 4, name: 'html' }, { glyph: 4, name: 'link' }, { glyph: 4, name: 'p' }, { glyph: 4, name: 'script js', value: 'script type="text/javascript"' }, { glyph: 4, name: 'span' }, { glyph: 4, name: 'title' } ]; intellisense5.setDeclarations(data); } });
